{ int:a, int: b, int: out } process1 = {
    out = a + b;
}
        
functype Process {
    int: a,
    int: b,
    int: out
};

Process process2 = {
    out = a + b;
}

int my_var;

process1(10, 20, my_var);
process2(10, 20, my_var); // same result
