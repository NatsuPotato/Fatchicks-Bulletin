Quark

Compiles to C. Low level with QoL. Memory safe. Everything is pass by reference.

// functions

functype DoSomething {
    int: a,
    int: b,
    int: out
};

DoSomething my_function = {
    out = a + b;
}

// abstract functions (no type)

{ int:a, int: b, int: out } my_abstract_function = {
    out = a + b;
}

int my_var;
my_function(10, 20, my_var);
my_abstract_function(10, 20, my_var); // same result
