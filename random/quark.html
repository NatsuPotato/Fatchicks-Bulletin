High level data processing language.


type MyType = {
    x: int,
    y: int
}
        
List my_list;

100 & MyType(10, 20) & "cats" & 20.0 & 10 & my_list;

for element in my_list
| String -> { print(element); }
| Int    -> { element /= 10;  }
| MyType -> { element = element.x;  }
| _      -> { pop element;    }
|;
