Quark

High level Document Object Model processing language.

* element = document.body;

match child in element {
    p {
        child.innerText = "Hello world!";
    }
    * {
        pop child;
    }
}
my_template(title, body) {
    <article>
        <h1>${ title }</h1>
        <p>${ body }</p>
    </article>
};

document.body push my_template("My Headline", "Lorem ipsum dolor sit amet.");
